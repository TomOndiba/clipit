 33:      * Create a version of a string for embedding in a URL
 34:      *
 35:      * @param string $string    A UTF-8 string
 36:      * @param string $separator The character to separate words with
 37:      * @return string
 38:      */
 39:     static public function urlize($string, $separator = '-') {
 40:         // Iñtërnâtiônàlizætiøn, AND 日本語!
 41: 
 42:         // try to force combined chars because the translit map and others expect it
 43:         if (self::hasNormalizerSupport()) {
 44:             $nfc = normalizer_normalize($string);
 45:             if (is_string($nfc)) {
 46:                 $string = $nfc;
 47:             }
 48:         }
 49:         // Internationalization, AND 日本語!
 50:         $string = self::transliterateAscii($string);
 51: 
 52:         // allow HTML tags in titles
 53:         $string = preg_replace('~<([a-zA-Z][^>]*)>~', ' $1 ', $string);
 54: 
 55:         // more substitutions
 56:         // @todo put these somewhere else
 57:         $string = strtr($string, array(
 58:             // currency
 59:             "\xE2\x82\xAC" /* € */ => ' E ',
 60:             "\xC2\xA3" /* £ */ => ' GBP ',
 61:         ));
 62: 
 63:         // remove all ASCII except 0-9a-zA-Z, hyphen, underscore, and whitespace
 64:         // note: "x" modifier did not work with this pattern.
 65:         $string = preg_replace('~['
 66:             . '\x00-\x08'  // control chars
 67:             . '\x0b\x0c'   // vert tab, form feed
 68:             . '\x0e-\x1f'  // control chars
 69:             . '\x21-\x2c'  // ! ... ,
 70:             . '\x2e\x2f'   // . slash
 71:             . '\x3a-\x40'  // : ... @
 72:             . '\x5b-\x5e'  // [ ... ^
 73:             . '\x60'       // `
 74:             . '\x7b-\x7f'  // { ... DEL
 75:             . ']~', '', $string);
 76:         $string = strtr($string, '', '');
 77: 
 78:         // internationalization, and 日本語!
 79:         // note: not using elgg_strtolower to keep this class portable
 80:         $string = is_callable('mb_strtolower')
 81:             ? mb_strtolower($string, 'UTF-8')
 82:             : strtolower($string);
 83: 
 84:         // split by ASCII chars not in 0-9a-zA-Z
 85:         // note: we cannot use [^0-9a-zA-Z] because that matches multibyte chars.
 86:         // note: "x" modifier did not work with this pattern.
 87:         $pattern = '~['
 88:             . '\x00-\x2f'  // controls ... slash
 89:             . '\x3a-\x40'  // : ... @
 90:             . '\x5b-\x60'  // [ ... `
 91:             . '\x7b-\x7f'  // { ... DEL
 92:             . ']+~x';
 93: 
 94:         // ['internationalization', 'and', '日本語']
 95:         $words = preg_split($pattern, $string, -1, PREG_SPLIT_NO_EMPTY);
 96: 
 97:         // ['internationalization', 'and', '%E6%97%A5%E6%9C%AC%E8%AA%9E']
 98:         $words = array_map('urlencode', $words);
 99: 
100:         // internationalization-and-%E6%97%A5%E6%9C%AC%E8%AA%9E
101:         return implode($separator, $words);
102:     }
