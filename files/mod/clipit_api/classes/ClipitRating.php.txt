/**
 * Class ClipitRating
 *
 */
class ClipitRating extends UBItem{
    /**
     * @const string Elgg entity subtype for this class
     */
    const SUBTYPE = "clipit_rating";

    public $target = 0;
    /**
     * @var int Overall rating opinionfrom 0 to 10
     */
    public $overall_rating = 0;
    /**
     * @var array Ratings about Tags used"
     */
    public $tag_rating_array = array();
    /**
     * @var array Ratings about Performance tips used"
     */
    public $performance_rating_array = array();


    protected function load_from_elgg($elgg_object){
        parent::load_from_elgg($elgg_object);
        $this->overall_rating = (int)$elgg_object->overall;
        $this->tag_rating_array = (array)$elgg_object->tag_rating_array;
        $this->performance_rating_array = (array)$elgg_object->performance_rating_array;
    }

    /**
     * @param ElggObject $elgg_object Elgg object instance to save Item to
     */
    protected function copy_to_elgg($elgg_object){
        parent::copy_to_elgg($elgg_object);
        $elgg_object->overall_rating = (int)$this->overall_rating;
        $elgg_object->tag_rating_array = (array)$this->tag_rating_array;
        $elgg_object->performance_rating_array = (array)$this->performance_rating_array;
    }

    static function add_tag_rating($rating_id, $tag_id, $is_used, $comment){
        $rating = new static($rating_id);
        $prop_value_array["tag_id"] = $tag_id;
        $prop_value_array["is_used"] = (bool)$is_used;
        $prop_value_array["comment"] = (string)$comment;
        $tag_rating_id = ClipitTagRating::create($prop_value_array);
        if(!empty($tag_rating_id){
            $rating->tag_rating_array[] = $tag_rating_id;
            return $rating->save();
        } else{
            return null;
        }
    }

    static function add_performance_rating($rating_id, $performance_id, $star_rating){
        $rating = new static($rating_id);
        $prop_value_array["performance_id"] = $performance_id;
        $prop_value_array["star_rating"] = (bool)$star_rating;
        $prop_value_array["comment"] = (string)$comment;
        $tag_rating_id = ClipitTagRating::create($prop_value_array);
        if(!empty($tag_rating_id){
            $rating->tag_rating_array[] = $tag_rating_id;
            return $rating->save();
        } else{
            return null;
        }
    }


}
