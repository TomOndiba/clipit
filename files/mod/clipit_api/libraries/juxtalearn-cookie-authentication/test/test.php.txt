// Utilities.
function _get($key, $default = NULL) {
    return isset($_GET[$key]) ? $_GET[$key] : $default;
}

function cookie_domain() {
    $host = _get('host') ? _get('host') : $_SERVER['HTTP_HOST'];
    if (preg_match('@localhost@', $host)) {
      return 'localhost';
    }
    else {
      return preg_replace('@^[^\.]+\.@', '.', $host);
    }
}
